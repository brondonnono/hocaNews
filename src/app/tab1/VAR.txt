VAR 
LIGNE TYPE LISTE
I,J,N NOMBRE
NEXTlIGNE TYPE LISTE
LIRE N
SI N>2
    LIGNE[1] = 1
    LIGNE[2] = 1
    POUR I DE 2 A N
        NEXTlIGNE[I] = 1
        ECRIRE I 
        POUR J DE 2 A I 
            NEXTlIGNE[J] = LIGNE[J-1]+LIGNE[J]
        FINPOUR
        NEXTlIGNE[I+1] = 1
        POUR J DE 2 A I+1
            LIGNE[J] = NEXTlIGNE[J]
            ECRIRE ";"
            ECRIRE LIGNE[J]
        FINPOUR
        ECRIRE ""
    FINPOUR
FINSI
